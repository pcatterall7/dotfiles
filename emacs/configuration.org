** Package Management

*** Setup Package Archives

Add MELPA to package sources and initialize the package system.

#+begin_src emacs-lisp
(require 'package)

*** Bootstrap use-package

Ensure =use-package= is installed, as we use it to manage all other packages.

(unless (package-installed-p 'use-package)

(require 'use-package)

* Editor Fundamentals

** Evil Mode (Vim Keybindings)

*** Evil Core

Evil mode brings vim's modal editing to Emacs. We configure it before loading to ensure proper integration.

#+begin_src emacs-lisp
(use-package evil
  :demand t
  :init
  ;; These settings must be set before evil loads
  (setq evil-want-integration t)        ; Enable evil integration with other packages
  (setq evil-want-keybinding nil)       ; Let evil-collection handle keybindings
  (setq evil-undo-system 'undo-redo)    ; Use Emacs 28+ undo system
  (setq evil-want-C-u-scroll t)         ; Ctrl-u scrolls up (vim behavior)
  :config
  (evil-mode 1))
#+end_src

*** Evil Collection

Evil-collection provides vim keybindings for many Emacs modes that don't come with Evil by default.

#+begin_src emacs-lisp
(use-package evil-collection

Go to the latest week list

#+begin_src emacs-lisp
  (defun my/narrow-to-week-list ()
    "Narrow to the first sub-heading under the top-level 'Week Lists' heading."
    (interactive)
    (widen)
    (goto-char (point-min))
    (if (re-search-forward "^\\* Week Lists$" nil t)
        (let ((subtree-end (save-excursion (org-end-of-subtree t t))))
          (if (re-search-forward "^\\*\\* " subtree-end t)
              (progn
                (beginning-of-line)
                (org-narrow-to-subtree)
  	      (org-show-subtree)
                (goto-char (point-min)))
            (message "No sub-heading found under 'Week Lists'")))
      (message "No 'Week Lists' heading found")))
#+end_src

Go to today
#+begin_src emacs-lisp
(defun my/narrow-to-today-working-memory ()
  "Narrow to the level 2 heading under 'Working Memory' that contains today's date."
  (interactive)
  (widen)
  (goto-char (point-min))
  (let ((today (format-time-string "%Y-%m-%d")))
    (if (re-search-forward "^\\* Working Memory$" nil t)
        (let ((subtree-end (save-excursion (org-end-of-subtree t t))))
          (if (re-search-forward (concat "^\\*\\*\\* .*" (regexp-quote today)) subtree-end t)
              (progn
                (beginning-of-line)
                (org-narrow-to-subtree)
                (goto-char (point-min)))
            (message "No heading with today's date (%s) found under 'Working Memory'" today)))
      (message "No 'Working Memory' heading found"))))
#+end_src

Zoom out one level
#+begin_src emacs-lisp
  (defun my/org-widen-one-level ()
  "Widen the current org narrowing by one level up in the tree."
  (interactive)
  (when (buffer-narrowed-p)
    (widen)
    (org-back-to-heading t)
    (when (org-up-heading-safe)
      (org-narrow-to-subtree))))
#+end_src

*** General.el Setup

General.el makes it easy to set up a Space leader key like in Vim or Doom Emacs.

#+begin_src emacs-lisp
  (use-package general
    :demand t
    :config
    (general-create-definer my-leader-def
      :states '(normal visual emacs)
      :keymaps 'override
      :prefix "SPC")

    (my-leader-def
      ;; File operations
      "f" '(:ignore t :which-key "files")
      "ff" '(find-file :which-key "find file")
      "fr" '(consult-recent-file :which-key "recent files")
      "fs" '(save-buffer :which-key "save file")
      "fS" '(write-file :which-key "save as")
      "fc" '(my/open-config :which-key "open emacs config file")

      ;; Buffer operations
      "b" '(:ignore t :which-key "buffers")
      "bb" '(consult-buffer :which-key "switch buffer")
      "bd" '(kill-current-buffer :which-key "kill buffer")
      "bn" '(next-buffer :which-key "next buffer")
      "bp" '(previous-buffer :which-key "previous buffer")

      ;; Window operations
      "w" '(:ignore t :which-key "windows")
      "ww" '(other-window :which-key "other window")
      "wo" '(delete-other-windows :which-key "delete other windows")
      "wd" '(delete-window :which-key "delete window")
      "wo" '(delete-other-windows :which-key "delete other windows")
      "ws" '(split-window-below :which-key "split horizontal")
      "wv" '(split-window-right :which-key "split vertical")

      ;; Search operations
      "s" '(:ignore t :which-key "search")
      "ss" '(consult-line :which-key "search line")
      "sg" '(consult-grep :which-key "grep")
      "sr" '(replace-string :which-key "replace string")
      "sR" '(replace-regexp :which-key "replace regexp")

      ;; Org mode
      "o" '(:ignore t :which-key "org")
      "oo" '(consult-outline :which-key "outline")
      "og" '(org-goto :which-key "go to")
      "on" '(org-narrow-to-subtree :which-key "narrow to subtree")
      "oW" '(widen :which-key "widen")
      "ow" '(my/org-widen-one-level :which-key "widen one level")
      "ot" '(my/narrow-to-today-working-memory :which-key "go to today")
      "oT" '(my/narrow-to-week-list :which-key "go to week list")
      "oi" '(org-insert-structure-template :which-key "insert block")
      "ol" '(org-insert-link :which-key "insert link")

      ;; GPTel
      "a" '(:ignore t :which-key "gptel")
      "ai" '(gptel :which-key "open gptel")

      ;; Utility
      "SPC" '(execute-extended-command :which-key "M-x")
      "q" '(:ignore t :which-key "quit")
      "qq" '(save-buffers-kill-terminal :which-key "quit emacs")
      "h" '(:ignore t :which-key "help")
      "hf" '(describe-function :which-key "describe function")
      "hv" '(describe-variable :which-key "describe variable")
      "hk" '(describe-key :which-key "describe key")))
#+end_src

#+begin_src emacs-lisp
  (add-to-list 'custom-theme-load-path "~/.emacs.d/themes/")

  (defun toggle-light-dark-theme ()
    "Automatically adjust theme based on macOS appearance."
    (interactive)
    (let ((appearance (shell-command-to-string "defaults read -g AppleInterfaceStyle")))
      (if (string-match-p "Dark" appearance)
	(progn
	  (mapc #'disable-theme custom-enabled-themes)
	  (load-theme 'modus-vivendi t))
	(progn
	  (mapc #'disable-theme custom-enabled-themes)
	  (load-theme 'modus-operandi t)))))
#+end_src
